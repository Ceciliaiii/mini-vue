# 组件卸载
根据不同节点类型（普通元素、Fragment、组件）执行对应的移除操作
```ts
const unmount = (vnode) => {
  const { shapeFlag } = vnode;

  // 处理Fragment节点（片段节点）
  if (vnode.type === Fragment) {
    unmountChildren(vnode.children); // 卸载所有子节点
  }
  
  // 处理组件节点
  else if (shapeFlag & ShapeFlags.COMPONENT) {
    unmount(vnode.component.subTree); // 卸载组件渲染的子树
  }

  // 处理普通元素节点
  else {
    hostRemove(vnode.el); // 直接移除真实DOM元素
  }
};
```
