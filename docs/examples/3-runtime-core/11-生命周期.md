# 生命周期

```ts
// 子组件中使用生命周期

const Child = {
  setup() {
    
    // 1. 注册“挂载前”钩子：组件DOM挂载前执行
    onBeforeMount(() => {
      console.log('组件即将挂载');
    });

    // 2. 注册“挂载后”钩子：组件DOM挂载后执行
    onMounted(() => {
      console.log('组件已挂载，可操作DOM');
    });

    // 3. 注册“更新前”钩子：组件状态更新前执行
    onBeforeUpdate(() => {
      console.log('组件即将更新');
    });

    // 4. 注册“更新后”钩子：组件状态更新后执行
    onUpdated(() => {
      console.log('组件已更新');
    });

    return () => h('div', '子组件');
  }
};
```
