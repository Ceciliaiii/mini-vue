# 异步组件
通过配置对象自定义加载状态、错误处理、超时等：
```ts
const AsyncComponent = defineAsyncComponent({
  // 加载函数（必选）：返回 Promise
  loader: () => import('./DataList.vue'),
  
  // 加载中显示的组件（可选）
  loadingComponent: {
    render() { return h('div', '加载中...'); }
  },
  
  // 加载失败显示的组件（可选）
  errorComponent: {
    render() { return h('div', '加载失败，请重试'); }
  },
  
  // 延迟显示加载组件（毫秒，可选）：避免短暂加载闪烁
  delay: 200, // 200ms 内加载完成则不显示 loading
  
  // 超时时间（毫秒，可选）：超过时间视为加载失败
  timeout: 5000, // 5秒未加载完成则显示 errorComponent
  
  // 错误回调（可选）：可手动重试
  onError(err, retry, fail) {
    // err：错误信息
    // retry：重试加载的函数
    // fail：标记为加载失败的函数
    console.error('加载失败:', err);
    // 示例：点击按钮重试
    if (window.confirm('加载失败，是否重试？')) {
      retry();
    } else {
      fail();
    }
  }
});
```
