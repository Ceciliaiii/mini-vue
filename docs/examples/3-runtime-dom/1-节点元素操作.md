# 节点元素操作

```html
<div id="app"></div>
<script type="module">

import {h, createRenderer, renderer} from "..."
import {renderOptions} from "./runtime-dom.js"

// h函数参数：元素type、元素属性、元素文本
let ele = h(
    "h1",
    { style: {color: 'red'}, class: "abc", onClick: () => alert(1) },
    "handsome czx"
)

// createRenderer 自己创建渲染器，用自己的渲染方式
const renderer = createRenderer(renderOptions)

// renderer 用内置的渲染器进行渲染
renderer.render(ele, app)

</script>


```
